const SHEET_NAME = 'Data'; const HEADERS = [ 'audit_id', 'audit_date', 'agent_name', 'audit_channel', 'property_name_score', 'mgmt_company_score', 'regional_mgr_score', 'email_score', 'website_score', 'office_hours_score', 'commission_score', 'locator_note_score', 'requirements_score', 'occupancy_score', 'special_coding_score', 'pricing_score', 'availability_score', 'notation_score', 'basic_raw', 'basic_weighted', 'locator_raw', 'locator_weighted', 'occupancy_raw', 'occupancy_weighted', 'floorplan_raw', 'floorplan_weighted', 'notation_raw', 'notation_weighted', 'overall_raw', 'overall_score_100', 'final_error_note' ]; const WEIGHTS = { basic: 50, locator: 20, occupancy: 10, floorplan: 10, notation: 10, }; function doGet(e) { try { const sheet = getOrCreateSheet_(); ensureHeaders_(sheet); const lastRow = sheet.getLastRow(); if (lastRow <= 1) { return jsonResponse_({ status: 'success', data: [], headers: HEADERS, }); } const values = sheet.getRange(1, 1, lastRow, HEADERS.length).getValues(); const rows = values.slice(1).map((row) => rowToObject_(HEADERS, row)); rows.sort((a, b) => { const da = String(a.audit_date || ''); const db = String(b.audit_date || ''); if (da !== db) return db.localeCompare(da); return String(b.audit_id || '').localeCompare(String(a.audit_id || '')); }); return jsonResponse_({ status: 'success', data: rows, headers: HEADERS, }); } catch (error) { return jsonResponse_({ status: 'error', message: error && error.message ? error.message : String(error), }); } } function doPost(e) { try { const sheet = getOrCreateSheet_(); ensureHeaders_(sheet); const payload = parseRequest_(e); const action = String(payload.action || '').trim().toLowerCase(); if (!action) { throw new Error('Missing action.'); } if (action === 'create') { const created = createRecord_(sheet, payload); return jsonResponse_({ status: 'success', message: 'Record created successfully.', data: created, }); } if (action === 'update') { const updated = updateRecord_(sheet, payload); return jsonResponse_({ status: 'success', message: 'Record updated successfully.', data: updated, }); } if (action === 'delete') { deleteRecord_(sheet, payload.audit_id); return jsonResponse_({ status: 'success', message: 'Record deleted successfully.', }); } throw new Error('Invalid action. Use create, update, or delete.'); } catch (error) { return jsonResponse_({ status: 'error', message: error && error.message ? error.message : String(error), }); } } function parseRequest_(e) { const result = {}; if (e && e.parameter) { Object.keys(e.parameter).forEach((key) => { result[key] = e.parameter[key]; }); } const contents = e && e.postData && e.postData.contents ? String(e.postData.contents) : ''; const contentType = e && e.postData && e.postData.type ? String(e.postData.type).toLowerCase() : ''; if (contents) { if (contentType.indexOf('application/json') !== -1) { const parsed = JSON.parse(contents || '{}'); if (parsed && typeof parsed === 'object') { if (parsed.data && typeof parsed.data === 'object') { Object.keys(parsed.data).forEach((key) => { result[key] = parsed.data[key]; }); } Object.keys(parsed).forEach((key) => { if (key !== 'data') result[key] = parsed[key]; }); } } else if (contentType.indexOf('application/x-www-form-urlencoded') !== -1 && (!e || !e.parameter || !Object.keys(e.parameter).length)) { contents.split('&').forEach((part) => { const pair = part.split('='); const key = decodeURIComponent((pair[0] || '').replace(/\+/g, ' ')); const value = decodeURIComponent((pair.slice(1).join('=') || '').replace(/\+/g, ' ')); if (key) result[key] = value; }); } } return result; } function createRecord_(sheet, input) { const record = sanitizeAndComputeRecord_(input); record.audit_id = generateAuditId_(sheet); const row = HEADERS.map((h) => record[h] !== undefined ? record[h] : ''); sheet.appendRow(row); return record; } function updateRecord_(sheet, input) { const auditId = String(input.audit_id || '').trim(); if (!auditId) throw new Error('audit_id is required for update.'); const lastRow = sheet.getLastRow(); if (lastRow < 2) throw new Error('No records found.'); const data = sheet.getRange(1, 1, lastRow, HEADERS.length).getValues(); const idIndex = HEADERS.indexOf('audit_id'); for (let i = 1; i < data.length; i++) { if (String(data[i][idIndex]) === auditId) { const existing = rowToObject_(HEADERS, data[i]); const merged = Object.assign({}, existing, input); const record = sanitizeAndComputeRecord_(merged); record.audit_id = auditId; const row = HEADERS.map((h) => record[h] !== undefined ? record[h] : ''); sheet.getRange(i + 1, 1, 1, HEADERS.length).setValues([row]); return record; } } throw new Error('Record not found: ' + auditId); } function deleteRecord_(sheet, auditId) { const id = String(auditId || '').trim(); if (!id) throw new Error('audit_id is required for delete.'); const lastRow = sheet.getLastRow(); if (lastRow < 2) throw new Error('No records found.'); const data = sheet.getRange(1, 1, lastRow, HEADERS.length).getValues(); const idIndex = HEADERS.indexOf('audit_id'); for (let i = 1; i < data.length; i++) { if (String(data[i][idIndex]) === id) { sheet.deleteRow(i + 1); return; } } throw new Error('Record not found: ' + id); } function sanitizeAndComputeRecord_(input) { const record = { audit_id: text_(input.audit_id), audit_date: normalizeDate_(input.audit_date), agent_name: text_(input.agent_name), audit_channel: text_(input.audit_channel), property_name_score: score_(input.property_name_score), mgmt_company_score: score_(input.mgmt_company_score), regional_mgr_score: score_(input.regional_mgr_score), email_score: score_(input.email_score), website_score: score_(input.website_score), office_hours_score: score_(input.office_hours_score), commission_score: score_(input.commission_score), locator_note_score: score_(input.locator_note_score), requirements_score: score_(input.requirements_score), occupancy_score: score_(input.occupancy_score), special_coding_score: score_(input.special_coding_score), pricing_score: score_(input.pricing_score), availability_score: score_(input.availability_score), notation_score: score_(input.notation_score), final_error_note: text_(input.final_error_note), }; if (!record.audit_date) throw new Error('audit_date is required.'); if (!record.agent_name) throw new Error('agent_name is required.'); return Object.assign(record, computeScores_(record)); } function computeScores_(r) { const basic_raw = num_(r.property_name_score) + num_(r.mgmt_company_score) + num_(r.regional_mgr_score) + num_(r.email_score) + num_(r.website_score) + num_(r.office_hours_score); const locator_raw = num_(r.commission_score) + num_(r.locator_note_score) + num_(r.requirements_score); const occupancy_raw = num_(r.occupancy_score) + num_(r.special_coding_score); const floorplan_raw = num_(r.pricing_score) + num_(r.availability_score); const notation_raw = num_(r.notation_score); const basic_weighted = scale_(basic_raw, 30, WEIGHTS.basic); const locator_weighted = scale_(locator_raw, 15, WEIGHTS.locator); const occupancy_weighted = scale_(occupancy_raw, 10, WEIGHTS.occupancy); const floorplan_weighted = scale_(floorplan_raw, 10, WEIGHTS.floorplan); const notation_weighted = scale_(notation_raw, 5, WEIGHTS.notation); const overall_raw = basic_raw + locator_raw + occupancy_raw + floorplan_raw + notation_raw; const overall_score_100 = basic_weighted + locator_weighted + occupancy_weighted + floorplan_weighted + notation_weighted; return { basic_raw: round2_(basic_raw), basic_weighted: round2_(basic_weighted), locator_raw: round2_(locator_raw), locator_weighted: round2_(locator_weighted), occupancy_raw: round2_(occupancy_raw), occupancy_weighted: round2_(occupancy_weighted), floorplan_raw: round2_(floorplan_raw), floorplan_weighted: round2_(floorplan_weighted), notation_raw: round2_(notation_raw), notation_weighted: round2_(notation_weighted), overall_raw: round2_(overall_raw), overall_score_100: round2_(overall_score_100), }; } function scale_(raw, maxRaw, maxWeighted) { if (!maxRaw) return 0; return round2_((num_(raw) / maxRaw) * maxWeighted); } function text_(v) { return String(v == null ? '' : v).trim(); } function num_(v) { const n = Number(v); return isFinite(n) ? n : 0; } function score_(v) { let n = Number(v); if (!isFinite(n)) n = 0; if (n < 0) n = 0; if (n > 5) n = 5; return round2_(n); } function round2_(n) { return Math.round((Number(n) + Number.EPSILON) * 100) / 100; } function normalizeDate_(value) { if (!value) return ''; if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value)) { return Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } const s = String(value).trim(); if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s; const d = new Date(s); if (!isNaN(d.getTime())) { return Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } return ''; } function generateAuditId_(sheet) { const now = new Date(); const base = Utilities.formatDate(now, Session.getScriptTimeZone(), 'yyyyMMdd-HHmmss'); const rand = Math.floor(100 + Math.random() * 900); const id = 'AUD-' + base + '-' + rand; const lastRow = sheet.getLastRow(); if (lastRow <= 1) return id; const ids = sheet.getRange(2, 1, lastRow - 1, 1).getValues().flat().map(String); if (ids.indexOf(id) === -1) return id; return 'AUD-' + base + '-' + Math.floor(1000 + Math.random() * 9000); } function getOrCreateSheet_() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); } return sheet; } function ensureHeaders_(sheet) { const currentMaxCols = sheet.getMaxColumns(); if (currentMaxCols < HEADERS.length) { sheet.insertColumnsAfter(currentMaxCols, HEADERS.length - currentMaxCols); } if (sheet.getLastRow() === 0) { sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]); styleHeader_(sheet); return; } const headerRange = sheet.getRange(1, 1, 1, HEADERS.length); const current = headerRange.getValues()[0]; let mismatch = false; for (let i = 0; i < HEADERS.length; i++) { if (String(current[i] || '') !== HEADERS[i]) { mismatch = true; break; } } if (mismatch) { headerRange.setValues([HEADERS]); } styleHeader_(sheet); } function styleHeader_(sheet) { const range = sheet.getRange(1, 1, 1, HEADERS.length); range .setFontWeight('bold') .setBackground('#dbeafe') .setFontColor('#1e3a8a') .setHorizontalAlignment('center'); sheet.setFrozenRows(1); } function rowToObject_(headers, row) { const obj = {}; headers.forEach((header, index) => { let value = row[index]; if (value instanceof Date) { value = Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } obj[header] = value; }); return obj; } function jsonResponse_(payload) { return ContentService .createTextOutput(JSON.stringify(payload)) .setMimeType(ContentService.MimeType.JSON); }
